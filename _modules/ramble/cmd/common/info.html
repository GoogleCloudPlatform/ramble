def print_text_info(obj):
    obj._verbosity = "long"
    color.cprint(str(obj))

def print_info(args):
    object_type = ramble.repository.ObjectTypes[args.type]
    obj_name = args.object
    obj = ramble.repository.get(obj_name, object_type=object_type)
    print_text_info(obj)
